CREATE TABLE locationsTypes (
	id SERIAL PRIMARY KEY,
	name VARCHAR(30) NOT NULL
);

CREATE TABLE locations (
	id SERIAL PRIMARY KEY,
	name VARCHAR(60) NOT NULL,
	type_id INT NOT NULL,
	FOREIGN KEY (type_id) REFERENCES locationsTypes(id)
);
